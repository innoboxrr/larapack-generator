<?php

namespace App\Models\Traits\Storage;

trait PascalCaseModelNameStorage
{

    public function createModel($request)
    {

        $camelCaseModelName = $this->create($request->only($this->creatable));

        return $camelCaseModelName;

    }

    public function updateModel($request)
    {
     
        $this->update($request->only($this->updatable));

        return $this;

    }

    public function updateModelMetas($request)
    {

        $this->update_metas($request, 'PascalCaseModelNameMeta', 'snake_case_model_name_id')->updatePayload();

        return $this;

    }

    public function deleteModel()
    {

        $this->delete();

    }

    public function restoreModel()
    {

        $this->restore();

    }

    public function forceDeleteModel()
    {

        abort(403);

        $this->forceDelete();
        
    }

}
